// Objetivo: detectar redeclaração de variável inteira em um escopo interno aninhado (bloco 'senao') quando já existe
//           uma variável com o mesmo nome declarada no escopo imediatamente externo (dentro do 'enquanto').
// Esperado: ERR_DECLARED na linha
//           'var variavel_inteira := inteiro com 2'
//           dentro do bloco 'senao', porque 'variavel_inteira' já tinha sido declarada antes no mesmo escopo de função.

principal -> inteiro :=
[
    enquanto (1)
    [
        var variavel_inteira := inteiro com 1

        se (1)
        [
            se (0)
            [
                
            ]
            senao
            [
                var variavel_inteira := inteiro com 2
            ]
        ]
    ]

    retorna 0 := inteiro
];
