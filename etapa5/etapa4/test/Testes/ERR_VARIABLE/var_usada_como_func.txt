//           argumentos entre parênteses).
// Esperado: ERR_VARIABLE na linha
//           'resultado_decimal := var_usada_como_func(contador, contador + 1)'
//           porque 'var_usada_como_func' foi declarada como variável (decimal) e não como função.

[

    var var_usada_como_func := decimal com 1.0
    var contador := inteiro com 1
    resultado_decimal := var_usada_como_func(contador, contador + 1)

    retorna 3.14 := decimal
];
